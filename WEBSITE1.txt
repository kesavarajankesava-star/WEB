#include<stdio.h>
#include<string.h>
struct employee
{
    int id;
    char name[20];
    float salary;

};
int main()
{
    struct employee stude;
    int id;
    char name[20];
    float salary;
    scanf("%d",&id);
    stude.id=id;
    scanf("%s",name);
    strcpy(stude.name,name);
    scanf("%f",&salary);
    stude.salary=salary;

    printf("employee id is:%d\n",stude.id);
    printf("name is:%s\n",stude.name);
    printf("salary is:%f",stude.salary);
}

#include<stdio.h>
#include<string.h>
struct employee
{
    int id;
    char name[20];
    float salary;

};
int main()
{
    struct employee stude;
    struct employee *s=&stude;
    int id;
    char name[20];
    float salary;
    scanf("%d",&id);
    s->id=id;
    scanf("%s",name);
    strcpy(s->name,name);
    scanf("%f",&salary);
    s->salary=salary;

    printf("employee id is:%d\n",s->id);

}

#include<stdio.h>
#include<stdlib.h>
#include<string.h>
struct employee
{
    int id;
    char name[20];
    float salary;
};
int main()
{
    struct employee stude[3];
    int itr;
    int id;
    char name[20];
    float salary;
    for(itr=0;itr<3;itr+=1)
    {
        scanf("%d",&id);
        stude[itr].id=id;
        scanf("%s",name);
        strcpy(stude[itr].name,name);
        scanf("%f",&salary);
        stude[itr].salary=salary;
    }
    for(itr=0;itr<3;itr+=1)

      printf("%d th employee's data :\n",itr+1);

      printf("name is:%s\n",stude[itr].name);

    }
}
